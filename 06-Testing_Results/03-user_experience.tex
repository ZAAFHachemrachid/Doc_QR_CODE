\subsection{User Satisfaction Survey}
\textcolor{TextBlack}{
    Survey results from 50 students and 10 faculty members:
    \begin{itemize}
        \item \result{95\% found the system easy to use}
        \item \result{92\% reported faster attendance process}
        \item \result{88\% preferred QR code over traditional methods}
        \item \result{94\% appreciated real-time attendance tracking}
        \item \result{89\% reported no technical difficulties}
    \end{itemize}
}
