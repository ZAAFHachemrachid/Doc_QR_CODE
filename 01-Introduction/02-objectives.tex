\section{Research Objectives}
\begin{infobox}
The primary objectives of this research focus on developing and implementing an efficient QR code-based attendance system.
\end{infobox}

\subsection{Primary Objectives}
\textcolor{TextBlack}{
    \begin{enumerate}
        \item \objective{Develop a secure and efficient QR code-based attendance system}
        \item \objective{Implement real-time attendance tracking and reporting}
        \item \objective{Reduce administrative workload in attendance management}
        \item \objective{Prevent proxy attendance through secure authentication}
        \item \objective{Enable data-driven decision making through analytics}
    \end{enumerate}
}

\subsection{Scope and Limitations}
\begin{notebox}
\textcolor{TextBlack}{
    This research focuses on:
    \begin{itemize}
        \item \method{Implementation within university environment}
        \item \method{Mobile-based QR code scanning}
        \item \method{Web-based administrative interface}
        \item \method{Data security and privacy considerations}
        \item \method{System scalability and performance analysis}
    \end{itemize}
}
\end{notebox}
